git clone https://github.com/Dede20231/LATAM.git
cd LATAM

# every time
mkdir DEST_REPO
cd DEST_REPO
git init

cp -r ../LATAM/* .
git status
git add .
git commit -m "Initial commit - imported source LATAM"
git remote add origin https://github.com/Dede20231/DEST_REPO.git
git remote -v

git branch -M main
git push -u origin main

git pull origin main --allow-unrelated-histories

git remote remove origin
